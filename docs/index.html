Set Option

var record = new viewRec({interval:200});

Start Recording

record.startRecord();

Stop Recording

record.stopRecord();

Play

record.play();

Stop Playing

record.stop();

set data Record

var data = [ [12,12,12,12],[12,12,12,12] ];
record.setData(data);

get data Record

record.getData();

Clear data Record

record.clearData();
